  const FarmerRegistrationStep3Screen({super.key, required this.isDarkMode});

  @override
  State<FarmerRegistrationStep3Screen> createState() => _FarmerRegistrationStep3ScreenState();
}

class _FarmerRegistrationStep3ScreenState extends State<FarmerRegistrationStep3Screen> {
  late bool isDarkMode;

  final TextEditingController dockNameController = TextEditingController();

  @override
  void initState() {
    super.initState();
    isDarkMode = widget.isDarkMode;
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: isDarkMode ? const Color(0xFF1a1a2e) : const Color(0xFFFFF8E7),
      body: SafeArea(
        child: SingleChildScrollView(
          child: Padding(
            padding: const EdgeInsets.symmetric(horizontal: 24.0),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const SizedBox(height: 20),

                Align(
                  alignment: Alignment.topRight,
                  child: GestureDetector(
                    onTap: () => setState(() => isDarkMode = !isDarkMode),
                    child: Container(
                      width: 50,
                      height: 50,
                      decoration: const BoxDecoration(color: Color(0xFFFF9800), shape: BoxShape.circle),
                      child: Icon(isDarkMode ? Icons.dark_mode : Icons.light_mode, color: Colors.white, size: 28),
                    ),
                  ),
                ),

                const SizedBox(height: 30),

                GestureDetector(
                  onTap: () => Navigator.pop(context),
                  child: Row(
                    children: [
                      Icon(Icons.arrow_back,
                          color: isDarkMode ? const Color(0xFF4CAF50) : const Color(0xFF2E7D32)),
                      const SizedBox(width: 8),
                      Text(
                        'Back to Login',
                        style: TextStyle(
                          fontSize: 16,
                          color: isDarkMode ? const Color(0xFF4CAF50) : const Color(0xFF2E7D32),
                          fontWeight: FontWeight.w600,
                        ),
                      ),
                    ],
                  ),
                ),

                const SizedBox(height: 40),

                Container(
                  padding: const EdgeInsets.all(24),
                  decoration: BoxDecoration(
                    color: isDarkMode ? const Color(0xFF2d2d44) : Colors.white,
                    borderRadius: BorderRadius.circular(16),
                    boxShadow: [
                      BoxShadow(color: Colors.black.withOpacity(0.1), blurRadius: 10, offset: const Offset(0, 4)),
                    ],
                  ),
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Text(
                        'Farmer Registration',
                        style: TextStyle(
                          fontSize: 28,
                          fontWeight: FontWeight.bold,
                          color: isDarkMode ? Colors.white : Colors.black,
                        ),
                      ),
                      const SizedBox(height: 4),
                      Text(
                        'Step 3 of 3',
                        style: TextStyle(fontSize: 14, color: isDarkMode ? Colors.white60 : Colors.black54),
                      ),

                      const SizedBox(height: 30),

                      Text(
                        'Dock Details',
                        style: TextStyle(
                          fontSize: 22,
                          fontWeight: FontWeight.bold,
                          color: isDarkMode ? Colors.white : Colors.black,
                        ),
                      ),
                      const SizedBox(height: 20),

                      _label('Dock Name', isDarkMode),
                      const SizedBox(height: 8),
                      TextField(
                        controller: dockNameController,
                        style: TextStyle(color: isDarkMode ? Colors.white : Colors.black),
                        decoration: InputDecoration(
                          hintText: 'Enter dock name',
                          hintStyle: TextStyle(color: isDarkMode ? Colors.white38 : Colors.black38),
                          filled: true,
                          fillColor: isDarkMode ? const Color(0xFF1a1a2e) : const Color(0xFFF5F5F5),
                          border: OutlineInputBorder(
                            borderRadius: BorderRadius.circular(12),
                            borderSide: BorderSide.none,
                          ),
                          contentPadding: const EdgeInsets.symmetric(horizontal: 16, vertical: 16),
                        ),
                      ),

                      const SizedBox(height: 18),

                      Container(
                        width: double.infinity,
                        padding: const EdgeInsets.all(14),
                        decoration: BoxDecoration(
                          color: const Color(0xFFFFF3CD),
                          borderRadius: BorderRadius.circular(12),
                        ),
                        child: const Text(
                          'Your registration will be reviewed by our team. You will receive an email notification once approved.',
                          style: TextStyle(fontSize: 14, color: Colors.black87, height: 1.4),
                        ),
                      ),

                      const SizedBox(height: 24),

                      Row(
                        children: [
                          SizedBox(
                            width: 120,
                            height: 50,
                            child: ElevatedButton(
                              onPressed: () => Navigator.pop(context),
                              style: ElevatedButton.styleFrom(
                                backgroundColor: const Color(0xFF4CAF50),
                                shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
                                elevation: 0,
                              ),
                              child: const Text(
                                'Back',
                                style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold, color: Colors.white),
                              ),
                            ),
                          ),
                          const Spacer(),
                          SizedBox(
                            width: 180,
                            height: 50,
                            child: ElevatedButton(
                              onPressed: () {
                                if (dockNameController.text.trim().isEmpty) {
                                  ScaffoldMessenger.of(context).showSnackBar(
                                    const SnackBar(content: Text('Please enter dock name')),
                                  );
                                  return;
                                }

                                Navigator.pushReplacement(
                                  context,
                                  MaterialPageRoute(
                                    builder: (_) => RegistrationSubmittedScreen(isDarkMode: isDarkMode),
                                  ),
                                );
                              },
                              style: ElevatedButton.styleFrom(
                                backgroundColor: const Color(0xFF4CAF50),
                                shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
                                elevation: 0,
                              ),
                              child: const Text(
                                'Submit Registration',
                                style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold, color: Colors.white),
                              ),
                            ),
                          ),
                        ],
                      ),
                    ],
                  ),
                ),

                const SizedBox(height: 40),
              ],
            ),
          ),
        ),
      ),
    );
  }

  @override
  void dispose() {
    dockNameController.dispose();
    super.dispose();
  }
}

// ============================================
// Small Helpers
// ============================================
Widget _label(String text, bool isDarkMode) {
  return Text(
    text,
    style: TextStyle(
      fontSize: 16,
      fontWeight: FontWeight.w600,
      color: isDarkMode ? Colors.white : Colors.black,
    ),
  );
}

Widget _input(TextEditingController controller, bool isDarkMode) {
  return TextField(
    controller: controller,
    style: TextStyle(color: isDarkMode ? Colors.white : Colors.black),
    decoration: InputDecoration(
      filled: true,
      fillColor: isDarkMode ? const Color(0xFF1a1a2e) : const Color(0xFFF5F5F5),
      border: OutlineInputBorder(
        borderRadius: BorderRadius.circular(12),
        borderSide: BorderSide.none,
      ),
      contentPadding: const EdgeInsets.symmetric(horizontal: 16, vertical: 16),
    ),
  );
}
