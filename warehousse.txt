// ===========================
// Simple Models (In-Memory)
// ===========================
class DeliveryRecord {
  final String farmer;
  final double quantityKg;
  final DateTime date;

  const DeliveryRecord({
    required this.farmer,
    required this.quantityKg,
    required this.date,
  });
}

class StorageRecord {
  final String farmer;
  final double quantityKg;
  final DateTime date;
  final String status; // 'Stored' or 'Retrieved'

  const StorageRecord({
    required this.farmer,
    required this.quantityKg,
    required this.date,
    required this.status,
  });
}

class CapacityHistoryItem {
  final String title;
  final DateTime date;
  final double valueKg;

  const CapacityHistoryItem({
    required this.title,
    required this.date,
    required this.valueKg,
  });
}

// ===========================
// Helpers
// ===========================
String fmtDate(DateTime d) {
  final dd = d.day.toString().padLeft(2, '0');
  final mm = d.month.toString().padLeft(2, '0');
  final yyyy = d.year.toString();
  return '$yyyy-$mm-$dd';
}

String fmtDateSlash(DateTime d) {
  final dd = d.day.toString().padLeft(2, '0');
  final mm = d.month.toString().padLeft(2, '0');
  final yyyy = d.year.toString();
  return '$dd/$mm/$yyyy';
}

Color borderGreen() => Colors.green.withAlpha((0.25 * 255).round());

// App root with global theme + global state
// ===========================
class WarehouseApp extends StatefulWidget {
  const WarehouseApp({super.key});

  @override
  State<WarehouseApp> createState() => _WarehouseAppState();
}

class _WarehouseAppState extends State<WarehouseApp> {
  bool isDarkMode = true;

  // Capacity state
  double maxCapacityKg = 10000;
  double currentCapacityKg = 7500;

  // Farmers
  final List<String> farmers = const [
    'John Farmer',
    'Mary Smith',
    'Robert Johnson',
    'Lisa Brown',
    'David Wilson',
  ];

  // Deliveries + Reports
  final List<DeliveryRecord> deliveries = [
    DeliveryRecord(
        farmer: 'John Farmer', quantityKg: 500, date: DateTime(2024, 1, 14)),
    DeliveryRecord(
        farmer: 'Mary Smith', quantityKg: 750, date: DateTime(2024, 1, 13)),
    DeliveryRecord(
        farmer: 'Robert Johnson', quantityKg: 600, date: DateTime(2024, 1, 12)),
  ];

  final List<StorageRecord> storageRecords = [
    StorageRecord(
        farmer: 'John Farmer',
        quantityKg: 500,
        date: DateTime(2024, 1, 14),
        status: 'Stored'),
    StorageRecord(
        farmer: 'Mary Smith',
        quantityKg: 750,
        date: DateTime(2024, 1, 13),
        status: 'Stored'),
    StorageRecord(
        farmer: 'Robert Johnson',
        quantityKg: 600,
        date: DateTime(2024, 1, 12),
        status: 'Stored'),
    StorageRecord(
        farmer: 'Lisa Brown',
        quantityKg: 450,
        date: DateTime(2024, 1, 11),
        status: 'Retrieved'),
    StorageRecord(
        farmer: 'David Wilson',
        quantityKg: 550,
        date: DateTime(2024, 1, 10),
        status: 'Stored'),
  ];

  final List<CapacityHistoryItem> capacityHistory = [
    CapacityHistoryItem(
        title: 'Capacity increased',
        date: DateTime(2024, 1, 10),
        valueKg: 10000),
    CapacityHistoryItem(
        title: 'Capacity adjusted',
        date: DateTime(2023, 12, 15),
        valueKg: 9500),
    CapacityHistoryItem(
        title: 'Initial capacity set',
        date: DateTime(2023, 11, 20),
        valueKg: 9000),
  ];

  void toggleTheme() => setState(() => isDarkMode = !isDarkMode);

  void updateMaxCapacity(double newMax) {
    setState(() {
      maxCapacityKg = newMax;
      capacityHistory.insert(
        0,
        CapacityHistoryItem(
          title: 'Capacity updated',
          date: DateTime.now(),
          valueKg: newMax,
        ),
      );
      currentCapacityKg =
          currentCapacityKg.clamp(0.0, maxCapacityKg).toDouble();
    });
  }

  void recordDelivery(DeliveryRecord record) {
    setState(() {
      deliveries.insert(0, record);

      storageRecords.insert(
        0,
        StorageRecord(
          farmer: record.farmer,
          quantityKg: record.quantityKg,
          date: record.date,
          status: 'Stored',
        ),
      );

      currentCapacityKg = (currentCapacityKg + record.quantityKg)
          .clamp(0.0, maxCapacityKg)
          .toDouble();
    });
  }

  @override
  Widget build(BuildContext context) {
    final lightTheme = ThemeData(
      brightness: Brightness.light,
      primaryColor: const Color(0xFF4CAF50),
      scaffoldBackgroundColor: const Color(0xFFFFF9E6),
      fontFamily: 'Roboto',
      appBarTheme: const AppBarTheme(
        backgroundColor: Colors.white,
        foregroundColor: Colors.black,
        elevation: 0,
      ),
      cardColor: Colors.white,
    );

    final darkTheme = ThemeData(
      brightness: Brightness.dark,
      primaryColor: const Color(0xFF4CAF50),
      scaffoldBackgroundColor: const Color(0xFF111827),
      fontFamily: 'Roboto',
      appBarTheme: const AppBarTheme(
        backgroundColor: Color(0xFF1F2937),
        foregroundColor: Colors.white,
        elevation: 0,
      ),
      cardColor: const Color(0xFF1F2937),
    );

    return MaterialApp(
      title: 'Warehouse Administration',
      debugShowCheckedModeBanner: false,
      theme: lightTheme,
      darkTheme: darkTheme,
      themeMode: isDarkMode ? ThemeMode.dark : ThemeMode.light,
      home: DashboardPage(
        isDarkMode: isDarkMode,
        onToggleTheme: toggleTheme,
        maxCapacityKg: maxCapacityKg,
        currentCapacityKg: currentCapacityKg,
        farmers: farmers,
        deliveries: deliveries,
        storageRecords: storageRecords,
        capacityHistory: capacityHistory,
        onUpdateMaxCapacity: updateMaxCapacity,
        onRecordDelivery: recordDelivery,
      ),
    );
  }
}

List<Widget> buildTopActions({
  required BuildContext context,
  required bool isDarkMode,
  required VoidCallback onToggleTheme,
}) {
  return [
    IconButton(
      icon: Icon(isDarkMode ? Icons.nightlight_round : Icons.wb_sunny),
      color: isDarkMode ? Colors.amber : Colors.orange,
      onPressed: onToggleTheme,
    ),

    // ✅ Popup menu يفتح كي نضغطو على الصورة
    PopupMenuButton<String>(
      tooltip: 'Account',
      offset: const Offset(0, 12),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(14)),
      onSelected: (value) {
        if (value == 'profile') {
          Navigator.push(
            context,
            MaterialPageRoute(builder: (_) => const ProfilePage()),
          );
        } else if (value == 'logout') {
          ScaffoldMessenger.of(context).showSnackBar(
            const SnackBar(content: Text('Logout (demo)')),
          );
        }
      },
      itemBuilder: (context) => const [
        PopupMenuItem(
          value: 'profile',
          child: Row(
            children: [
              Icon(Icons.person, size: 18),
              SizedBox(width: 10),
              Text('Profile'),
            ],
          ),
        ),
        PopupMenuItem(
          value: 'logout',
          child: Row(
            children: [
              Icon(Icons.logout, size: 18, color: Colors.red),
              SizedBox(width: 10),
              Text('Logout', style: TextStyle(color: Colors.red)),
            ],
          ),
        ),
      ],
      child: const Padding(
        padding: EdgeInsets.all(8.0),
        child: CircleAvatar(
          child: Text('100 × 100', style: TextStyle(fontSize: 8)),
        ),
      ),
    ),
  ];
}

class WeeklyCapacityChart extends StatelessWidget {
  const WeeklyCapacityChart({super.key});

  @override
  Widget build(BuildContext context) {
    final cardColor = Theme.of(context).cardColor;

    return Container(
      margin: const EdgeInsets.symmetric(horizontal: 16),
      padding: const EdgeInsets.all(20),
      decoration: BoxDecoration(
        color: cardColor,
        borderRadius: BorderRadius.circular(16),
        border: Border.all(color: borderGreen(), width: 2),
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          const Text('Weekly Capacity Usage',
              style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold)),
          const SizedBox(height: 20),
          SizedBox(
            height: 250,
            child: BarChart(
              BarChartData(
                alignment: BarChartAlignment.spaceAround,
                maxY: 100,
                barTouchData: BarTouchData(enabled: true),
                titlesData: FlTitlesData(
                  show: true,
                  bottomTitles: AxisTitles(
                    sideTitles: SideTitles(
                      showTitles: true,
                      getTitlesWidget: (value, meta) {
                        const days = [
                          'Mon',
                          'Tue',
                          'Wed',
                          'Thu',
                          'Fri',
                          'Sat',
                          'Sun'
                        ];
                        final i = value.toInt();
                        if (i >= 0 && i < days.length) {
                          return Text(days[i],
                              style: const TextStyle(color: Color(0xFF9CA3AF)));
                        }
                        return const SizedBox.shrink();
                      },
                    ),
                  ),
                  leftTitles: AxisTitles(
                    sideTitles: SideTitles(
                      showTitles: true,
                      reservedSize: 40,
                      getTitlesWidget: (value, meta) {
                        return Text(value.toInt().toString(),
                            style: const TextStyle(color: Color(0xFF9CA3AF)));
                      },
                    ),
                  ),
                  topTitles: const AxisTitles(
                      sideTitles: SideTitles(showTitles: false)),
                  rightTitles: const AxisTitles(
                      sideTitles: SideTitles(showTitles: false)),
                ),
                gridData: FlGridData(
                  show: true,
                  drawVerticalLine: false,
                  getDrawingHorizontalLine: (value) => FlLine(
                    color: Colors.grey.withAlpha((0.2 * 255).round()),
                    strokeWidth: 1,
                    dashArray: const [5, 5],
                  ),
                ),
                borderData: FlBorderData(show: false),
                barGroups: [
                  _bar(0, 65),
                  _bar(1, 72),
                  _bar(2, 68),
                  _bar(3, 75),
                  _bar(4, 78),
                  _bar(5, 53),
                  _bar(6, 45),
                ],
              ),
            ),
          ),
        ],
      ),
    );
  }

  BarChartGroupData _bar(int x, double y) {
    return BarChartGroupData(
      x: x,
      barRods: [
        BarChartRodData(
          toY: y,
          color: const Color(0xFF4CAF50),
          width: 30,
          borderRadius: const BorderRadius.vertical(top: Radius.circular(6)),
        )
      ],
    );
  }
}

                      width: double.infinity,
                      height: 52,
                      child: ElevatedButton(
                        style: ElevatedButton.styleFrom(
                          backgroundColor: const Color(0xFF4CAF50),
                          shape: RoundedRectangleBorder(
                              borderRadius: BorderRadius.circular(12)),
                        ),
                        onPressed: () {
                          final farmer = selectedFarmer;
                          final qty = double.tryParse(qtyCtrl.text.trim());

                          if (farmer == null || farmer.isEmpty) {
                            ScaffoldMessenger.of(context).showSnackBar(
                                const SnackBar(content: Text('Select a farmer')));
                            return;
                          }
                          if (qty == null || qty <= 0) {
                            ScaffoldMessenger.of(context).showSnackBar(const SnackBar(
                                content: Text('Enter a valid quantity')));
                            return;
                          }

                          widget.onRecordDelivery(
                            DeliveryRecord(
                                farmer: farmer,
                                quantityKg: qty,
                                date: selectedDate),
                          );

                          qtyCtrl.clear();
                          ScaffoldMessenger.of(context).showSnackBar(
                              const SnackBar(content: Text('Delivery recorded!')));
                          setState(() {});
                        },
                        child: const Text('Record Delivery',
                            style: TextStyle(
                                fontSize: 16, fontWeight: FontWeight.bold)),
                      ),
                    ),
                  ]),
            ),
            Container(
              margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
              padding: const EdgeInsets.all(20),
              decoration: BoxDecoration(
                color: cardColor,
                borderRadius: BorderRadius.circular(16),
                border: Border.all(color: borderGreen(), width: 2),
              ),
              child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    const Text('Recent Deliveries',
                        style: TextStyle(
                            fontSize: 18, fontWeight: FontWeight.bold)),
                    const SizedBox(height: 14),
                    ...widget.deliveries.take(6).map((d) {
                      return Column(
                        children: [
                          Row(
                            children: [
                              Expanded(
                                child: Column(
                                    crossAxisAlignment: CrossAxisAlignment.start,
                                    children: [
                                      Text(d.farmer,
                                          style: const TextStyle(
                                              fontWeight: FontWeight.bold)),
                                      const SizedBox(height: 2),
                                      Text(fmtDate(d.date),
                                          style: const TextStyle(
                                              color: Color(0xFF9CA3AF))),
                                    ]),
                              ),
                              Text('${d.quantityKg.toStringAsFixed(0)} kg',
                                  style: const TextStyle(
                                      fontWeight: FontWeight.bold)),
                            ],
                          ),
                          const SizedBox(height: 12),
                          const Divider(height: 1),
                          const SizedBox(height: 12),
                        ],
                      );
                    }),
                  ]),
            ),
          ],
        ),
      ),
    );
  }
}

  final void Function(DeliveryRecord) onRecordDelivery;

        .where((r) => r.status == 'Retrieved')
        .fold<double>(0, (s, r) => s + r.quantityKg);

        ),

        isDarkMode: isDarkMode,
        onToggleTheme: onToggleTheme,
        maxCapacityKg: maxCapacityKg,
        currentCapacityKg: currentCapacityKg,
        farmers: farmers,
        deliveries: deliveries,
        storageRecords: storageRecords,
        capacityHistory: capacityHistory,
        onUpdateMaxCapacity: onUpdateMaxCapacity,
        onRecordDelivery: onRecordDelivery,
      ),
      body: SingleChildScrollView(
        padding: const EdgeInsets.only(bottom: 24),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Padding(
              padding: const EdgeInsets.fromLTRB(16, 16, 16, 8),

                children: [
                  const Expanded(
                    child: Text('Storage Reports',
                        style: TextStyle(
                            fontSize: 28, fontWeight: FontWeight.bold)),
                  ),
                  ElevatedButton.icon(

                      backgroundColor: const Color(0xFFF59E0B),

                          content: Text('Export Report (demo)')));

                ],
              ),
            ),
            Container(
              margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
              padding: const EdgeInsets.all(20),
              decoration: BoxDecoration(
                color: cardColor,
                borderRadius: BorderRadius.circular(16),
                border: Border.all(color: borderGreen(), width: 2),
              ),
              child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,

                            fontSize: 18, fontWeight: FontWeight.bold)),

                            flex: 2,

                                style: TextStyle(color: Color(0xFF9CA3AF)))),

                                style: TextStyle(color: Color(0xFF9CA3AF)))),

                            flex: 2,
